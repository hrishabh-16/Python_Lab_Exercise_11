import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv("./lok Shabha Result 2019.csv")

df_15_rows = df.head(15)
Candidate_Vote_Share = df_15_rows['Candidate Vote Share %'].tolist()
Winning_Margin = df_15_rows['Winning Margin %'].tolist()

plt.plot(Candidate_Vote_Share, Winning_Margin)
plt.title("Line Graph")
plt.xlabel("Candidate Vote Share %")
plt.ylabel("Winning Margin %")
plt.grid(True)

# Show the graph
plt.show()
